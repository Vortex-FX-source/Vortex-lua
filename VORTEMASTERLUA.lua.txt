function OrionLib:MakeWindow(cfg)
    local Window = {}
    function Window:MakeTab(tcfg)
        local tabName = type(tcfg) == "table" and tcfg.Name or tcfg
        local TabBtn = Instance.new("TextButton", Sidebar)
        TabBtn.Size = UDim2.new(1, 0, 0, 35)
        TabBtn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
        TabBtn.Text = tabName
        TabBtn.TextColor3 = Color3.fromRGB(200, 200, 200)
        TabBtn.Font = Enum.Font.GothamSemibold
        Instance.new("UICorner", TabBtn).CornerRadius = UDim.new(0, 6)

        local Page = Instance.new("ScrollingFrame", ContentArea)
        Page.Size = UDim2.new(1, 0, 1, 0)
        Page.BackgroundTransparency = 1
        Page.ScrollBarThickness = 2
        Page.Visible = false
        Instance.new("UIListLayout", Page).Padding = UDim.new(0, 8)

        if not FirstTab then FirstTab = Page; Page.Visible = true; TabBtn.TextColor3 = AccentColor end

        TabBtn.MouseButton1Click:Connect(function()
            for _, c in pairs(ContentArea:GetChildren()) do if c:IsA("ScrollingFrame") then c.Visible = false end end
            for _, c in pairs(Sidebar:GetChildren()) do if c:IsA("TextButton") then c.TextColor3 = Color3.fromRGB(200, 200, 200) end end
            Page.Visible = true
            TabBtn.TextColor3 = AccentColor
        end)

        local Tab = {}
        function Tab:AddToggle(d)
            local ToggleBtn = Instance.new("TextButton", Page)
            ToggleBtn.Size = UDim2.new(1, -10, 0, 35)
            ToggleBtn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
            ToggleBtn.Text = "  " .. d.Name
            ToggleBtn.TextXAlignment = Enum.TextXAlignment.Left
            ToggleBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
            Instance.new("UICorner", ToggleBtn).CornerRadius = UDim.new(0, 6)
            local Indicator = Instance.new("Frame", ToggleBtn)
            Indicator.Size = UDim2.new(0, 15, 0, 15)
            Indicator.Position = UDim2.new(1, -25, 0.5, -7.5)
            Indicator.BackgroundColor3 = d.Default and AccentColor or Color3.fromRGB(60, 60, 60)
            Instance.new("UICorner", Indicator).CornerRadius = UDim.new(1, 0)
            local state = d.Default or false
            ToggleBtn.MouseButton1Click:Connect(function()
                state = not state
                Indicator.BackgroundColor3 = state and AccentColor or Color3.fromRGB(60, 60, 60)
                pcall(d.Callback, state)
            end)
            local ret = {}
            function ret:Set(val) state = val; Indicator.BackgroundColor3 = state and AccentColor or Color3.fromRGB(60, 60, 60); pcall(d.Callback, state) end
            return ret
        end
        function Tab:AddButton(d) CreateButton(Page, d.Name, d.Callback) end
        function Tab:AddSection(d) CreateLabel(Page, "--- " .. d.Name .. " ---", Color3.fromRGB(150,150,150)) end
        function Tab:AddLabel(text) return CreateLabel(Page, text, Color3.fromRGB(200,200,200)) end
        function Tab:AddParagraph(d) return CreateLabel(Page, type(d)=="string" and d or d[1], Color3.fromRGB(200,200,200)) end
        function Tab:AddTextbox(d) CreateInput(Page, d.Name, d.Callback) end
        function Tab:AddSlider(d)
            local frame = Instance.new("Frame", Page)
            frame.Size = UDim2.new(1, -10, 0, 45)
            frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
            Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 6)
            local lbl = Instance.new("TextLabel", frame)
            lbl.Size = UDim2.new(1, -10, 0, 20)
            lbl.Position = UDim2.new(0, 5, 0, 0)
            lbl.BackgroundTransparency = 1
            lbl.Text = d.Name .. ": " .. tostring(d.Default)
            lbl.TextColor3 = Color3.fromRGB(255,255,255)
            lbl.TextXAlignment = Enum.TextXAlignment.Left
            local barBg = Instance.new("TextButton", frame)
            barBg.Size = UDim2.new(1, -10, 0, 10)
            barBg.Position = UDim2.new(0, 5, 0, 25)
            barBg.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
            barBg.Text = ""
            local fill = Instance.new("Frame", barBg)
            fill.Size = UDim2.new((d.Default-d.Min)/(d.Max-d.Min), 0, 1, 0)
            fill.BackgroundColor3 = AccentColor
            local dragging = false
            barBg.MouseButton1Down:Connect(function() dragging = true end)
            UserInputService.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then dragging = false end end)
            RunService.RenderStepped:Connect(function()
                if dragging then
                    local mousePos = UserInputService:GetMouseLocation().X
                    local percent = math.clamp((mousePos - barBg.AbsolutePosition.X) / barBg.AbsoluteSize.X, 0, 1)
                    fill.Size = UDim2.new(percent, 0, 1, 0)
                    local val = math.floor(d.Min + ((d.Max - d.Min) * percent))
                    lbl.Text = d.Name .. ": " .. tostring(val)
                    pcall(d.Callback, val)
                end
            end)
        end
        function Tab:AddDropdown(d)
            local opts = d.Options or {}
            local current = 1
            local btn = CreateButton(Page, d.Name .. " : " .. tostring(opts[current] or ""), function() end)
            btn.MouseButton1Click:Connect(function()
                current = current + 1
                if current > #opts then current = 1 end
                local sel = opts[current]
                btn.Text = d.Name .. " : " .. tostring(sel)
                pcall(d.Callback, sel)
            end)
            local ret = {}
            function ret:Refresh(newOpts) opts = newOpts; btn.Text = d.Name .. " : " .. tostring(opts[1] or "") end
            return ret
        end
        return Tab
    end
    return Window
end

end
